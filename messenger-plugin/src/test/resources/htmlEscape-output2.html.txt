  function installSearchField() {
    Element.cleanWhitespace('userPanel');
    Element.cleanWhitespace('inputIdF');
    $('inputIdF').firstChild.id = 'headerSearchField';
    $('userPanel').innerHTML += &quot;&lt;span class='search'&gt;&quot; + $('inputIdF').innerHTML + &quot;&lt;/span&gt;&quot;;
    $('inputIdF').innerHTML ='';
    $('headerSearchField').onkeypress = function(event) {
      if (BS_search_timer) clearTimeout(BS_search_timer);
      var url = BS_search_getUrl();
      if (event.keyCode == Event.KEY_RETURN) {
        if(event.shiftKey) {
          document.location.href = url;
        } else {
          BS_search_showPopup(url);
        }
      } else if (event.keyCode == Event.KEY_ESC) {
        return false;
      } else {
        BS_search_timer = setTimeout(function () { BS_search_showPopup(BS_search_getUrl()); }, 700);
      }
    }.bindAsEventListener();
  }

  BS_search_getUrl = function() {
    return '/teamcity/searchResults.html?query='
        + encodeURIComponent($('headerSearchField').value)
        + '&amp;buildTypeId=';
  };

      afterShowFunc: function(popup) {
      }
    }).showPopupNearElement('headerSearchField');
  }
